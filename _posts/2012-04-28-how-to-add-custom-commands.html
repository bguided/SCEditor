Custom command demo

Code behind it

To add the HTML command, use the $.sceditor.command.set function:

{% highlight javascript %} $.sceditor.command.set( "headers", { exec: function(caller) { var editor = this, content = $("
"), clickFunc = function (e) { editor.execCommand("formatblock", ""); editor.closeDropDown(true); e.preventDefault(); }; for (var i=1; i<= 6; i++) { content.append( $('Heading ' + i + '') .data('sceditor-hsize', i) .click(clickFunc)); } editor.createDropDown(caller, "header-picker", content); }, tooltip: "Format Headers" } ); {% endhighlight %}

Then add the commands CSS:

{% highlight css %} .sceditor-button-headers div { background: url('/images/headers-button.png'); } .sceditor-header-option { display: block; cursor: pointer; font-size: 14px; text-decoration: none; color: #222; } .sceditor-header-option:hover { background: #eee; } {% endhighlight %}

Then, if using BBCode output, add the BBCode commands using the $.sceditorBBCodePlugin.bbcode.set function:

{% highlight javascript %} $.sceditorBBCodePlugin.bbcode .set("h1", { tags: { h1: null }, format: "[h1]{0}[/h1]", html: "

{0}

" }) .set("h2", { tags: { h2: null }, format: "[h2]{0}[/h2]", html: "

{0}

" }) .set("h3", { tags: { h3: null }, format: "[h3]{0}[/h3]", html: "

{0}

" }) .set("h4", { tags: { h4: null }, format: "[h4]{0}[/h4]", html: "

{0}

" }) .set("h5", { tags: { h5: null }, format: "[h5]{0}[/h5]", html: "
{0}
" }) .set("h6", { tags: { h6: null }, format: "[h6]{0}[/h6]", html: "
{0}
" }); {% endhighlight %}

Finally add the command to the toolbar of the editor:

{% highlight javascript %} $("#demo-cust-cmd").sceditorBBCodePlugin({ toolbar: "headers|source", }); {% endhighlight %}

See the documentation for more details on the above functions and CSS.
